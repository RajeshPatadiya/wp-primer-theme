367: /**
368:  * Alter the `<title>` tag based on what is being viewed.
369:  *
370:  * @filter wp_title
371:  * @global int $page
372:  * @global int $paged
373:  * @since  1.0.0
374:  *
375:  * @param  string $title Page title.
376:  * @param  string $sep   Title separator.
377:  *
378:  * @return string Return the page title.
379:  */
380: function primer_wp_title( $title, $sep ) {
381: 
382:     if ( is_feed() ) {
383: 
384:         return $title;
385: 
386:     }
387: 
388:     global $page, $paged;
389: 
390:     // Add the blog name.
391:     $title .= get_bloginfo( 'name', 'display' );
392: 
393:     // Add the blog description for the home/front page.
394:     $site_description = get_bloginfo( 'description', 'display' );
395: 
396:     if ( $site_description && ( is_home() || is_front_page() ) ) {
397: 
398:         $title .= sprintf(
399:             ' %s %s',
400:             $sep,
401:             $site_description
402:         );
403: 
404:     }
405: 
406:     // Add a page number if necessary.
407:     if ( ( $paged >= 2 || $page >= 2 ) && ! is_404() ) {
408: 
409:         $title .= sprintf(
410:             ' %s %s',
411:             $sep,
412:             sprintf(
413:                 /* translators: page number */
414:                 esc_html__( 'Page %d', 'primer' ),
415:                 max( $paged, $page )
416:             )
417:         );
418: 
419:     }
420: 
421:     return $title;
422: 
423: }
